# GET /users
curl -Method GET "http://localhost:8000/users"

# POST /users
curl -Method POST "http://localhost:8000/users" -Headers @{ "Content-Type"="application/json" } -Body '{ "name": "1234", "email": "opa@example.com" }'

# GET /products
curl -Method GET "http://localhost:8000/products"

# POST /products  
curl -Method POST "http://localhost:8000/products" -Headers @{ "Content-Type"="application/json" } -Body '{ "name": "Golf", "price": 11000, "stock": 3 }'

# GET /products/1
curl -Method GET "http://localhost:8000/products/1"

# PATCH /products/2/stock
curl -Method PATCH "http://localhost:8000/products/1/stock" -Headers @{ "Content-Type"="application/json" } -Body '{ "stock": 15 }'

# PATCH /products/4
curl -Method PATCH "http://localhost:8000/products/1" -Headers @{ "Content-Type"="application/json" } -Body '{ "price": 15000 }'

curl -Method POST "http://localhost:8000/orders" -Headers @{ "Content-Type"="application/json" } -Body '{ "userId": 1, "products": [ { "productId": 1, "quantity": 1, "price": 15000 } ], "paymentMethod": "credit_card" }'
